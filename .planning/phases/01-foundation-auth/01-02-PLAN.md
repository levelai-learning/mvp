1. **Create `database/schema.sql`** — all tables with RLS enabled immediately after creation. Include exactly these tables from the RESEARCH.md code examples, with these exact columns:

```sql
-- user_roles: read by custom_access_token_hook
create table public.user_roles (
  user_id uuid primary key references auth.users(id) on delete cascade,
  role text not null check (role in ('student', 'parent', 'teacher')),
  created_at timestamptz not null default now()
);
alter table public.user_roles enable row level security;

-- profiles: display name, preferences from onboarding
create table public.profiles (
  user_id uuid primary key references auth.users(id) on delete cascade,
  display_name text not null,
  preferred_name text,
  avatar_emoji text,
  avatar_bg text,
  learning_style text,
  dark_mode boolean not null default false,
  dyslexic_font boolean not null default false,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);
alter table public.profiles enable row level security;

-- classrooms: one per teacher (D-10)
create table public.classrooms (
  id uuid primary key default gen_random_uuid(),
  teacher_id uuid not null references auth.users(id),
  name text not null,
  created_at timestamptz not null default now(),
  unique(teacher_id)
);
alter table public.classrooms enable row level security;

-- classroom_students: enrollment join table
create table public.classroom_students (
  classroom_id uuid not null references public.classrooms(id),
  student_id uuid not null references auth.users(id),
  enrolled_at timestamptz not null default now(),
  primary key (classroom_id, student_id)
);
alter table public.classroom_students enable row level security;

-- parent_invites: single-use invite codes (D-03)
create table public.parent_invites (
  id uuid primary key default gen_random_uuid(),
  code text not null unique,
  student_id uuid not null references auth.users(id),
  created_by uuid not null references auth.users(id),
  expires_at timestamptz not null default (now() + interval '7 days'),
  used_at timestamptz,
  used_by uuid references auth.users(id)
);
alter table public.parent_invites enable row level security;

-- parent_student_links: durable parent-child relationship
create table public.parent_student_links (
  id uuid primary key default gen_random_uuid(),
  parent_id uuid not null references auth.users(id),
  student_id uuid not null references auth.users(id),
  linked_at timestamptz not null default now(),
  invite_id uuid references public.parent_invites(id),
  unique(parent_id, student_id)
);
alter table public.parent_student_links enable row level security;

-- consent_records: COPPA compliance tracking (AUTH-05)
create table public.consent_records (
  id uuid primary key default gen_random_uuid(),
  student_id uuid not null references auth.users(id),
  parent_id uuid not null references auth.users(id),
  consent_type text not null default 'coppa_data_collection',
  scope text not null default 'educational_data',
  verification_method text not null default 'email',
  granted_at timestamptz not null default now(),
  granted_by_user_id uuid not null references auth.users(id),
  revoked_at timestamptz,
  ip_address inet
);
alter table public.consent_records enable row level security;

-- audit_log: FERPA-ready event tracking (AUTH-07)
create table public.audit_log (
  id bigserial primary key,
  user_id uuid references auth.users(id),
  action text not null,
  resource_type text not null,
  resource_id text,
  ip_address inet,
  user_agent text,
  created_at timestamptz not null default now()
);
alter table public.audit_log enable row level security;
```

2. **Create `database/rls-policies.sql`** — MUST use `(select auth.uid())` wrapper (not bare `auth.uid()`) for the 94% performance improvement per Supabase docs. Include ALL policies from RESEARCH.md Pattern 3 plus:

   - `user_roles`: users can read their own role only
   - `profiles`: users can read/update their own profile only
   - `classrooms`: teacher can CRUD their own classroom
   - `classroom_students`: teacher sees their roster; student sees own enrollment
   - `parent_invites`: teacher can create/read invites they created; parent can read unexpired unused invites (for code validation)
   - `parent_student_links`: parent sees their own links; student sees links to them
   - `consent_records`: parent can read consent records they granted
   - `audit_log`: NO user reads — `using (false)` policy; only service role can read

3. **Create `database/auth-hook.sql`** — the `custom_access_token_hook` function from RESEARCH.md Pattern 4 exactly as specified. Include:
   - The function body reading from `public.user_roles`
   - `grant execute on function public.custom_access_token_hook to supabase_auth_admin;`
   - `revoke execute on function public.custom_access_token_hook from authenticated, anon;`
   - A comment reminding the user: "After running this SQL, go to Supabase Dashboard > Authentication > Hooks > Custom Access Token and point it at this function."

4. **Create `database/audit-triggers.sql`** — create Postgres trigger functions that auto-log writes:
   - A generic trigger function `log_data_change()` that inserts into `audit_log` with `action = TG_OP` (INSERT/UPDATE/DELETE), `resource_type = TG_TABLE_NAME`, `resource_id = NEW.id or OLD.id`, `user_id = (select auth.uid())`
   - Attach triggers to: `profiles`, `classrooms`, `classroom_students`, `parent_student_links`, `consent_records`
   - Each trigger fires AFTER INSERT OR UPDATE OR DELETE
